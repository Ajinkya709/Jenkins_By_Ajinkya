pipeline {
    agent { label "node" }

    environment {
        APP_NAME     = "simple-python-app"
        DOCKER_IMAGE = "ajinkya0709/simple-python-app:latest"
    }

    stages {
        stage('Checkout Code') {
            steps {
                git branch: 'main', url: 'https://github.com/Ajinkya709/Jenkins_By_Ajinkya.git'
                echo "âœ… Code cloned successfully!"
            }
        }

        stage('Build Docker Image') {
            steps {
                echo "ðŸš€ Building Docker image for ${APP_NAME}..."
                sh "docker build -t ${DOCKER_IMAGE} ."
                echo "âœ… Docker image built successfully: ${DOCKER_IMAGE}"
            }
        }

        stage('Push to Docker Hub') {
            steps {
                echo "ðŸ“¤ Pushing image to Docker Hub..."
                withCredentials([usernamePassword(credentialsId: 'dockerhubcred', usernameVariable: 'DOCKER_USER', passwordVariable: 'DOCKER_PASS')]) {
                    sh """
                        echo "$DOCKER_PASS" | docker login -u "$DOCKER_USER" --password-stdin
                        docker push ${DOCKER_IMAGE}
                    """
                }
                echo "âœ… Image pushed to Docker Hub successfully!"
            }
        }

        stage('Test App') {
            steps {
                echo "ðŸ§ª Running unit tests..."
                sh "pytest --maxfail=1 --disable-warnings -q || true"
                echo "âœ… Tests executed!"
            }
        }

        stage('Deploy App') {
            steps {
                echo "ðŸš€ Deploying container..."
                sh """
                    docker rm -f ${APP_NAME} || true
                    docker run -d --name ${APP_NAME} -p 5000:5000 ${DOCKER_IMAGE}
                """
                echo "âœ… Application deployed successfully and running on port 5000!"
            }
        }
    }

    post {
        always {
            echo "ðŸ§¹ Cleaning up after build..."
        }
    }
}
